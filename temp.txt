        end
      when 'flot'
        if tipo_op2 == 'flot'
          return 'flot'
        else
          raise "Error"
        end
      when 'string'
        if tipo_op2 == 'string' or tipo_op2 == 'char'
          return 'string'
        else
          raise "Error"
        end
      when 'char'
        if tipo_op2 == 'char'
          return 'string'
        else
          raise "Error"
        end
      when 'bol'
        raise "Error"

    when '-'
    when '/'
    when '*'

    when '<', '>'

    when '='
    when '!='
